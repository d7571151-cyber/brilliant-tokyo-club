/** * BRILLIANT TOKYO CLUB - セッション予約管理 */ function doPost(e) { const data = JSON.parse(e.postData.contents); const action = data.action; if (action === 'book') { return handleBooking(data); } else if (action === 'getSessions') { return getSessionsWithCounts(); } return ContentService.createTextOutput( JSON.stringify({ result: 'error', message: 'unknown action' }) ).setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return getSessionsWithCounts(); } function handleBooking(data) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const bookingSheet = ss.getSheetByName('予約'); const sessionSheet = ss.getSheetByName('セッション'); const sessionId = String(data.sessionId); const name = data.name; const contact = data.contact; const sessionData = sessionSheet.getDataRange().getValues(); let capacity = 0; let sessionFound = false; for (let i = 1; i < sessionData.length; i++) { if (String(sessionData[i][0]) === sessionId) { capacity = Number(sessionData[i][4]); sessionFound = true; break; } } if (!sessionFound) { return ContentService.createTextOutput( JSON.stringify({ result: 'error', message: 'session not found' }) ).setMimeType(ContentService.MimeType.JSON); } const bookingData = bookingSheet.getDataRange().getValues(); let currentCount = 0; for (let i = 1; i < bookingData.length; i++) { if (String(bookingData[i][1]) === sessionId) { currentCount++; } } if (currentCount >= capacity) { return ContentService.createTextOutput( JSON.stringify({ result: 'full', message: '満員です' }) ).setMimeType(ContentService.MimeType.JSON); } bookingSheet.appendRow([new Date(), sessionId, name, contact]); const notifyEmail = "daichiyamazaki.jpn@gmail.com"; MailApp.sendEmail(notifyEmail, "【BRILLIANT TOKYO CLUB】新規予約: " + name, "セッションID: " + sessionId + "\nお名前: " + name + "\n連絡先: " + contact); return ContentService.createTextOutput( JSON.stringify({ result: 'success', newCount: currentCount + 1, capacity: capacity }) ).setMimeType(ContentService.MimeType.JSON); } function getSessionsWithCounts() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const bookingSheet = ss.getSheetByName('予約'); const sessionSheet = ss.getSheetByName('セッション'); const sessionData = sessionSheet.getDataRange().getValues(); const bookingData = bookingSheet.getDataRange().getValues(); const counts = {}; for (let i = 1; i < bookingData.length; i++) { const sid = String(bookingData[i][1]); counts[sid] = (counts[sid] || 0) + 1; } const sessions = []; for (let i = 1; i < sessionData.length; i++) { const row = sessionData[i]; sessions.push({ id: String(row[0]), date: String(row[1]), time: formatTimeValue(row[2]), location: String(row[3]), capacity: Number(row[4]), count: counts[row[0]] || 0 }); } return ContentService.createTextOutput( JSON.stringify({ result: 'success', sessions: sessions }) ).setMimeType(ContentService.MimeType.JSON); } function formatTimeValue(value) { if (Object.prototype.toString.call(value) === '[object Date]') { const hours = String(value.getHours()).padStart(2, '0'); const minutes = String(value.getMinutes()).padStart(2, '0'); return hours + ':' + minutes; } return String(value); }
